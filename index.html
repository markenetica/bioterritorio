        function calculateCarbon() {

            const transport = parseFloat(document.getElementById('transport').value) || 0;

            const publicTransport = parseFloat(document.getElementById('publicTransport').value) || 0;

            const electricity = parseFloat(document.getElementById('electricity').value) || 0;

            const gas = parseFloat(document.getElementById('gas').value) || 0;

            const waste = parseFloat(document.getElementById('waste').value) || 0;

            const water = parseFloat(document.getElementById('water').value) || 0;

            const domesticFlights = parseFloat(document.getElementById('domesticFlights').value) || 0;

            const internationalFlights = parseFloat(document.getElementById('internationalFlights').value) || 0;

            const dietFactor = parseFloat(document.getElementById('diet').value) || 3.3;

            const shoppingFactor = parseFloat(document.getElementById('shopping').value) || 0.3;

            

            // Factores de emisión para Colombia

            const transportFactor = 0.00021;

            const publicTransportFactor = 0.00008;

            const electricityFactor = 0.000126;

            const gasFactor = 0.00185;

            const wasteFactor = 0.00046;

            const waterFactor = 0.00023;

            const domesticFlightFactor = 0.2;

            const internationalFlightFactor = 1.5;

            

            // Cálculos anuales

            const transportEmissions = (transport * 12 * transportFactor) / 1000;

            const publicTransportEmissions = (publicTransport * 12 * publicTransportFactor) / 1000;

            const electricityEmissions = (electricity * 12 * electricityFactor) / 1000;

            const gasEmissions = (gas * 12 * gasFactor) / 1000;

            const wasteEmissions = (waste * 12 * wasteFactor) / 1000;

            const waterEmissions = (water * 12 * waterFactor) / 1000;

            const flightEmissions = (domesticFlights * domesticFlightFactor) + (internationalFlights * internationalFlightFactor);

            const dietEmissions = dietFactor;

            const shoppingEmissions = shoppingFactor;

            

            const total = transportEmissions + publicTransportEmissions + electricityEmissions + 

                         gasEmissions + wasteEmissions + waterEmissions + flightEmissions + 

                         dietEmissions + shoppingEmissions;

            

            const trees = Math.round(total / 0.021);

            

            document.getElementById('carbonTotal').textContent = total.toFixed(2);

            document.getElementById('trees').textContent = trees;

            

            let comparison = "";

            if (total < 1.6) {

                comparison = "• ¡Excelente! Tu huella está por debajo del promedio colombiano 🎉";

            } else if (total < 4.5) {

                comparison = "• Tu huella está entre el promedio colombiano y mundial 📊";

            } else {

                comparison = "• Tu huella está por encima del promedio mundial. ¡Es hora de actuar! 🌍";

            }

            document.getElementById('comparison').textContent = comparison;

            

            document.getElementById('result').style.display = 'block';

            

            setTimeout(() => {

                document.getElementById('result').scrollIntoView({ behavior: 'smooth', block: 'center' });

            }, 100);

        }
